        function createParticles() {

            const particlesContainer = document.getElementById('particles');

            for (let i = 0; i < 30; i++) {

                const particle = document.createElement('div');

                particle.className = 'particle';

                particle.style.left = Math.random() * 100 + '%';

                particle.style.animationDelay = Math.random() * 6 + 's';

                particle.style.animationDuration = (Math.random() * 4 + 4) + 's';

                particlesContainer.appendChild(particle);

            }

        }


        function populateResearchGrid(projects = researchProjects) {

            const grid = document.getElementById('researchGrid');

            grid.innerHTML = '';

            

            projects.forEach(project => {

                const card = document.createElement('div');

                card.className = 'research-card';

                card.setAttribute('data-category', project.category);

                

                let mediaGallery = '';

                if (project.media && project.media.length > 0) {

                    mediaGallery = '<div class="media-gallery">';

                    project.media.forEach(mediaUrl => {

                        if (mediaUrl.includes('streamable.com')) {

                            const videoId = mediaUrl.split('/').pop();

                            mediaGallery += `<div class="media-item"><div class="video-wrapper"><iframe allow="fullscreen" allowfullscreen src="https://streamable.com/e/${videoId}?" style="border:none;"></iframe></div></div>`;

                        } else if (mediaUrl.endsWith('.mp4') || mediaUrl.endsWith('.webm')) {

                            mediaGallery += `<div class="media-item"><video controls><source src="${mediaUrl}" type="video/mp4"></video></div>`;

                        } else {

                            mediaGallery += `<div class="media-item"><img src="${mediaUrl}" alt="Project media"></div>`;

                        }

                    });

                    mediaGallery += '</div>';

                }

                

                card.innerHTML = `

                    <div class="researcher-name">${project.name}</div>

                    <div class="research-program">${project.program}</div>

                    <div class="research-description">${project.research}</div>

                    <div class="visualization-need">${project.visualization}</div>

                    ${project.status ? `<div style="margin-top: 15px; padding: 8px; background: ${project.status.includes('✅') ? '#e8f5e8' : '#fff3cd'}; border-radius: 6px; font-weight: bold; color: ${project.status.includes('✅') ? '#155724' : '#856404'}; text-align: center;">${project.status}</div>` : ''}

                    ${mediaGallery}

                `;

                

                grid.appendChild(card);

            });

        }

        function switchTab(tabName) {

            const projectsTab = document.getElementById('projectsTab');

            const participantsTab = document.getElementById('participantsTab');

            const tabButtons = document.querySelectorAll('.controls .filter-button');

            

            tabButtons.forEach(btn => btn.classList.remove('active'));

            event.target.classList.add('active');

            

            if (tabName === 'projects') {

                projectsTab.style.display = 'block';

                participantsTab.style.display = 'none';

            } else if (tabName === 'participants') {

                projectsTab.style.display = 'none';

                participantsTab.style.display = 'block';

            }

        }


        function filterParticipants(category) {

            const buttons = document.querySelectorAll('.sub-controls .filter-button');

            const cards = document.querySelectorAll('.participant-card');

            

            buttons.forEach(btn => btn.classList.remove('active'));

            event.target.classList.add('active');

            

            cards.forEach(card => {

                const cardCategory = card.getAttribute('data-category');

                const programType = card.getAttribute('data-program-type');

                let shouldShow = false;

                

                if (category === 'all') {

                    shouldShow = true;

                } else if (category === 'matched') {

                    shouldShow = card.innerHTML.includes('Matched');

                } else if (category === 'phd') {

                    shouldShow = programType === 'phd';

                } else if (category === 'masters') {

                    shouldShow = programType === 'masters';

                } else {

                    shouldShow = cardCategory === category;

                }

                

                if (shouldShow) {

                    card.style.display = 'block';

                    card.style.opacity = '0';

                    setTimeout(() => card.style.opacity = '1', 100);

                } else {

                    card.style.display = 'none';

                }

            });

        }

        function highlightCard(card) {

            const allCards = document.querySelectorAll('.stat-card');

            allCards.forEach(c => c.style.background = 'rgba(255, 255, 255, 0.95)');

            card.style.background = 'linear-gradient(45deg, #667eea, #764ba2)';

            card.style.color = 'white';

            

            setTimeout(() => {

                card.style.background = 'rgba(255, 255, 255, 0.95)';

                card.style.color = 'inherit';

            }, 2000);

        }
